Data Loading

мқҙ л¬ём„ңлҠ” л¶„мӮ° н•ҷмҠөм—җм„ңмқҳ лҚ°мқҙн„° лЎңл”© нҢЁн„ҙмқ„ м •лҰ¬н•©лӢҲлӢӨ. ліё мҝЎл¶ҒмқҖ м¶”мІң(user-item interaction) мӢңлӮҳлҰ¬мҳӨлҘј к°Җм •н•©лӢҲлӢӨ.

ліё мҝЎл¶Ғмқҳ кё°ліё м„ нғқ: MovieLens 25M

мҝЎл¶Ғ м…Җл“Өмқҳ н•ҷмҠө мӢңк°„мқҖ 15л¶„ мқҙлӮҙлЎң м ңн•ңн•©лӢҲлӢӨ. л¶„мӮ° н•ҷмҠө нҢЁн„ҙ мһҗмІҙм—җ м§‘мӨ‘н•ҳл©ҙм„ңлҸ„ val/lossк°Җ мқҳлҜё мһҲкІҢ к°җмҶҢн•ҳлҠ” н•ҷмҠөмқ„ ліҙм—¬мЈјкё° мң„н•ҙ MovieLens 25M (GroupLens, ~250MB) мӢӨлҚ°мқҙн„°лҘј мӮ¬мҡ©н•©лӢҲлӢӨ. н•©м„ұ лҚ°мқҙн„°лҘј м“°л©ҙ лқјлІЁмқҙ лһңлҚӨмқҙлқј metricмқҙ мӣҖм§Ғмқҙм§Җ м•Ҡкё° л•Ңл¬ёмһ…лӢҲлӢӨ.

мһҗм„ён•ң лҚ°мқҙн„° мҠӨнӮӨл§ҲмҷҖ implicit feedback ліҖнҷҳ к·ңм№ҷмқҖ concepts-recommender-baseline.mdм—җ мһҲмҠөлӢҲлӢӨ. н•өмӢ¬мқ„ н•ң н‘ңлЎң мҡ”м•Ҫн•ҳл©ҙ лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

лӢЁкі„ мІҳлҰ¬
лӢӨмҡҙлЎңл“ң ml-25m.zip вҶ’ UC Volumeм—җ unzip
лқјлІЁ ліҖнҷҳ rating >= 4 вҶ’ positive (label=1)
negative sampling positive лӢ№ 1к°ң, uniform лһңлҚӨ movie (мқҙлҜё ліё кІғ м ңмҷё м—Ҷмқҙ к°„лӢЁнһҲ)
index remap userId, movieId вҶ’ 0-based contiguous (embedding lookupмҡ©)
split н–ү лӢЁмң„ random 10% вҶ’ val
м ҖмһҘ train/, val/ к°Ғк°Ғ shard-XXXXX.parquet л¬¶мқҢмңјлЎң UC Volumeм—җ

лӢӨмҡҙлЎңл“ң + м••м¶• н•ҙм ң

лЁјм Җ driverмқҳ лЎңм»¬ л””мҠӨнҒ¬м—җ zipмқ„ л°ӣм•„ н’үлӢҲлӢӨ.

import os
import urllib.request
import zipfile

ML25M_URL = "https://files.grouplens.org/datasets/movielens/ml-25m.zip"
local_zip = "/local_disk0/ml-25m.zip"
extract_root = "/local_disk0/ml-25m-extract"

urllib.request.urlretrieve(ML25M_URL, local_zip)
with zipfile.ZipFile(local_zip) as zf:
    zf.extractall(extract_root)
ratings_csv = os.path.join(extract_root, "ml-25m", "ratings.csv")

/local_disk0мқҖ л…ёл“ң лЎңм»¬ л””мҠӨнҒ¬мһ…лӢҲлӢӨ. UC Volumeм—җ л°”лЎң zipмқ„ н’Җм–ҙлҸ„ лҗҳм§Җл§Ң zip н•ҙм ң I/Oк°Җ лҠҗлҰ¬лҜҖлЎң, driver лЎңм»¬ л””мҠӨнҒ¬м—җм„ң н‘ј л’Ө UC VolumeмңјлЎң мҳ®кё°лҠ” нҺёмқҙ л№ лҰ…лӢҲлӢӨ.

лқјлІЁ ліҖнҷҳ + negative sampling + remap

implicit feedback ліҖнҷҳкіј мқёлҚұмҠӨ remap, negative samplingмқ„ н•ң лІҲм—җ мІҳлҰ¬н•©лӢҲлӢӨ.

import numpy as np
import pandas as pd

ratings = pd.read_csv(ratings_csv)  # ~600MB in pandas
pos = ratings.loc[ratings["rating"] >= 4.0, ["userId", "movieId"]].copy()
pos["label"] = np.float32(1.0)

# dense remap: userId, movieId вҶ’ 0..N-1 contiguous index
unique_users = np.sort(ratings["userId"].unique())
unique_items = np.sort(ratings["movieId"].unique())
user_to_idx = {u: i for i, u in enumerate(unique_users)}
item_to_idx = {m: i for i, m in enumerate(unique_items)}
pos["user_id"] = pos["userId"].map(user_to_idx).astype(np.int64)
pos["item_id"] = pos["movieId"].map(item_to_idx).astype(np.int64)

# negative sampling: positive лӢ№ 1к°ң, uniform лһңлҚӨ movie
rng = np.random.default_rng(0)
n_pos = len(pos)
neg = pd.DataFrame({
    "user_id": pos["user_id"].to_numpy(),
    "item_id": rng.integers(0, len(unique_items), size=n_pos, dtype=np.int64),
    "label": np.zeros(n_pos, dtype=np.float32),
})

interactions = pd.concat([pos[["user_id", "item_id", "label"]], neg], ignore_index=True)
interactions = interactions.sample(frac=1.0, random_state=0).reset_index(drop=True)
n_users = len(unique_users)
n_items = len(unique_items)

negative sampling м •нҷ•лҸ„м—җ кҙҖн•ң л©”лӘЁмһ…лӢҲлӢӨ. мң„ мҪ”л“ңлҠ” positiveлҘј вҖңмқҙлҜё ліё movieвҖқм—җм„ң м ңмҷён•ҳм§Җ м•ҠлҠ” лӢЁмҲң uniform мғҳн”Ңл§Ғмһ…лӢҲлӢӨ. ML-25M densityлҠ” 25M / (162K Г— 59K) вүҲ 0.26%лЎң л§Өмҡ° sparseн•ҳкё° л•Ңл¬ём—җ negativeлЎң лҪ‘мқҖ movieк°Җ мҡ°м—°нһҲ к·ё userмқҳ positiveмқј нҷ•лҘ мқҖ л¬ҙмӢңн•  мҲҳ мһҲмҠөлӢҲлӢӨ. м •л°Җн•ң м¶”мІң нҸүк°Җк°Җ н•„мҡ”н•ҳл©ҙ userлі„ positive setмқ„ м ңмҷён•ҳлҠ” rejection samplingмқ„ мҚЁм•ј н•ҳм§Җл§Ң, ліё мҝЎл¶ҒмқҖ л¶„мӮ° н•ҷмҠө л©”м»ӨлӢү лҚ°лӘЁк°Җ лӘ©м ҒмқҙлҜҖлЎң лӢЁмҲңнҷ”н•©лӢҲлӢӨ.

shard лӢЁмң„ parquet write (лҢҖмҡ©лҹү м•Ҳм „)

м „мІҙ ~25M н–үмқ„ driverм—җ н•ң лІҲм—җ мҳ¬лҰ¬лҠ” кІғмқҖ л¬ҙкІҒмҠөлӢҲлӢӨ. н–ү лӢЁмң„ shardлЎң мһҳлқј train/кіј val/м—җ лӮҳлҲ„м–ҙ м“°л©ҙ н•ҷмҠө л…ёнҠёл¶Ғмқҙ pq.read_table(files)лЎң л¬¶м–ҙ мқҪмқ„ мҲҳ мһҲмҠөлӢҲлӢӨ.

import os
import pyarrow as pa
import pyarrow.parquet as pq

def write_sharded(df, out_dir, val_ratio=0.1, rows_per_shard=1_000_000, seed=0):
    rng = np.random.default_rng(seed)
    os.makedirs(os.path.join(out_dir, "train"), exist_ok=True)
    os.makedirs(os.path.join(out_dir, "val"), exist_ok=True)
    for shard_idx, start in enumerate(range(0, len(df), rows_per_shard)):
        chunk = df.iloc[start : start + rows_per_shard]
        is_val = rng.random(size=len(chunk)) < val_ratio
        for split, mask in [("train", ~is_val), ("val", is_val)]:
            sub = chunk.loc[mask, ["user_id", "item_id", "label"]]
            if len(sub) == 0:
                continue
            table = pa.table({
                "user_id": sub["user_id"].to_numpy(),
                "item_id": sub["item_id"].to_numpy(),
                "label": sub["label"].to_numpy(),
            })
            pq.write_table(table, os.path.join(out_dir, split, f"shard-{shard_idx:05d}.parquet"))

мғқм„ұ кІ°кіј л””л үнҶ лҰ¬ кө¬мЎ°лҠ” лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

н•ҷмҠө л…ёнҠёл¶Ғмқҳ лҚ°мқҙн„° лЎңл”©

н•ҷмҠө мҪ”л“ңлҠ” train/кіј val/мқ„ к°Ғк°Ғ мқҪм–ҙ TensorDatasetмқ„ л§Ңл“ӯлӢҲлӢӨ.

import os
import pyarrow.parquet as pq
import torch
from torch.utils.data import TensorDataset


def load_split(split_dir):
    files = sorted(
        os.path.join(split_dir, f) for f in os.listdir(split_dir) if f.endswith(".parquet")
    )
    table = pq.read_table(files, columns=["user_id", "item_id", "label"])
    return TensorDataset(
        torch.from_numpy(table.column("user_id").to_numpy()),
        torch.from_numpy(table.column("item_id").to_numpy()),
        torch.from_numpy(table.column("label").to_numpy()),
    )

train_dataset = load_split(os.path.join(DATA_DIR, "train"))
val_dataset = load_split(os.path.join(DATA_DIR, "val"))

DDPм—җм„ңлҠ” torch.utils.data.distributed.DistributedSamplerлЎң rankлі„лЎң мһҗлҸҷ л¶„н• н•©лӢҲлӢӨ.

from torch.utils.data import DataLoader
from torch.utils.data.distributed import DistributedSampler

train_sampler = DistributedSampler(train_dataset, num_replicas=world_size, rank=global_rank, drop_last=True)
loader = DataLoader(train_dataset, batch_size=BATCH_SIZE, sampler=train_sampler, num_workers=2, pin_memory=True)

л©”лӘЁлҰ¬ к°Җмқҙл“ң

pq.read_tableмқҖ м „мІҙ splitмқ„ RAMм—җ мҳ¬лҰҪлӢҲлӢӨ. ML-25Mмқҳ train split(~22M н–ү, м»¬лҹј 3к°ң int/float)мқҖ м•Ҫ 600MBмқҙлҜҖлЎң driverмҷҖ к°Ғ worker GPU л…ёл“ңм—җ м¶©л¶„нһҲ л“Өм–ҙк°‘лӢҲлӢӨ. лҚ” нҒ° лҚ°мқҙн„°м—җм„ңлҠ” pq.ParquetDataset + lazy __getitem__ лҳҗлҠ” petastorm нҢЁн„ҙмқ„ кі л Өн•©лӢҲлӢӨ.

Databricks кІҪлЎң л№„көҗ вҖ” м–ҙл””м—җ лҚ°мқҙн„°лҘј л‘җлҠ”к°Җ

л¶„мӮ° н•ҷмҠөм—җм„ң лҚ°мқҙн„° кІҪлЎң м„ нғқмқҖ throughputкіј м•Ҳм •м„ұм—җ м§ҒкІ°лҗ©лӢҲлӢӨ. нӣ„ліҙлі„ нҠ№м„ұмқ„ н•ңлҲҲм—җ л№„көҗн•ҳл©ҙ лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

кІҪлЎң driver м ‘к·ј worker м ‘к·ј мҳҒмҶҚм„ұ read м„ұлҠҘ к¶ҢмһҘ мҡ©лҸ„
/local_disk0/... вң… вң… (к°Ғ л…ёл“ңмқҳ лЎңм»¬ SSD) нҒҙлҹ¬мҠӨн„° мў…лЈҢ мӢң мҶҢмӢӨ к°ҖмһҘ л№ лҰ„ (NVMe) мһ„мӢң unzip, scratch space
DBFS (/dbfs/...) вң… вң… мҳҒмҶҚ мӨ‘к°„ (FUSE мҳӨлІ„н—Өл“ң) л Ҳкұ°мӢң вҖ” мӢ к·ң мӮ¬мҡ© л№„к¶ҢмһҘ
UC Volume (/Volumes/.../) вң… вң… мҳҒмҶҚ, UC governance мӨ‘к°„ (cloud object store) ліё мҝЎл¶Ғ кё°ліё
Cloud direct (s3://, abfss://) API нҳём¶ң API нҳём¶ң мҳҒмҶҚ к°ҖмһҘ л№ лҘё cloud read (UC Volume мҡ°нҡҢ) TB+ лҚ°мқҙн„°, throughput мөңмҡ°м„ 

/local_disk0мқҳ л…ёл“ңлі„ л¶„лҰ¬

/local_disk0мқҖ к°Ғ л…ёл“ңмқҳ лЎңм»¬ SSDмһ…лӢҲлӢӨ. driverмқҳ /local_disk0кіј workerмқҳ /local_disk0мқҖ м„ңлЎң лӢӨлҘё л””мҠӨнҒ¬лқјлҠ” м җмқ„ кё°м–өн•ҙм•ј н•©лӢҲлӢӨ. л”°лқјм„ң лӢӨмқҢкіј к°ҷмқҖ нҠ№м„ұмқҙ л”°лқјмҳөлӢҲлӢӨ.

  • driverм—җм„ң unzipн•ҳкі  к°ҷмқҖ /local_disk0м—җ лҚ°мқҙн„°лҘј л‘җл©ҙ driverл§Ң н•ҷмҠөн•ҳлҠ” 1Г—1, 1Г—Nм—җм„ңлҠ” л№ лҘҙкІҢ лҸҷмһ‘н•©лӢҲлӢӨ.
  • multi-node(MГ—N) н•ҷмҠөмқҖ workerк°Җ driverмқҳ /local_disk0мқ„ ліј мҲҳ м—Ҷкё° л•Ңл¬ём—җ UC VolumeмңјлЎң мҳ®кІЁм•ј н•©лӢҲлӢӨ.
  • нҒҙлҹ¬мҠӨн„° restartлӮҳ auto-termination мӢң лӘЁл“  л…ёл“ңмқҳ /local_disk0мқҙ н•Ёк»ҳ мҶҢмӢӨлҗ©лӢҲлӢӨ.

ліё мҝЎл¶Ғмқҳ 01-data_prep.ipynbк°Җ /local_disk0м—җ unzipн•ң л’Ө UC Volumeм—җ shard parquetлЎң мҳ®кё°лҠ” мқҙмң к°Җ м—¬кё°м—җ мһҲмҠөлӢҲлӢӨ. driver-only мһ‘м—…мқҖ л№ лҘё лЎңм»¬мқ„ м“°кі , н•ҷмҠө лҚ°мқҙн„°лҠ” workerм—җм„ңлҸ„ ліҙмқҙлҠ” UC Volumeм—җ л‘җлҠ” л¶„м—…мһ…лӢҲлӢӨ.

UC Volume read м„ұлҠҘ

UC VolumeмқҖ лӮҙл¶Җм ҒмңјлЎң cloud object store(S3/ADLS/GCS)лҘј л§ҲмҡҙнҠён•ң кІғмқҙлҜҖлЎң лӢӨмқҢкіј к°ҷмқҖ м„ұлҠҘ нҠ№м„ұмқ„ ліҙмһ…лӢҲлӢӨ.

  • мІ« readлҠ” cloud latencyк°Җ к·ёлҢҖлЎң лҚ”н•ҙм ё мҲҳмӢӯ ms мҲҳмӨҖмһ…лӢҲлӢӨ.
  • к°ҷмқҖ л…ёл“ңм—җм„ң к°ҷмқҖ нҢҢмқјмқ„ лӢӨмӢң мқҪмңјл©ҙ Databricks Runtimeмқҳ disk cacheк°Җ SSDм—җ м ҖмһҘлҗҳм–ҙ л‘җ лІҲм§ё readл¶Җн„° л№Ёлқјм§‘лӢҲлӢӨ.
  • л…ёл“ң к°„м—җлҠ” cacheк°Җ кіөмң лҗҳм§Җ м•ҠмңјлҜҖлЎң multi-nodeм—җм„ң лӘЁл“  rankк°Җ к°ҷмқҖ train splitмқ„ мқҪмңјл©ҙ к°Ғ workerк°Җ лҸ…лҰҪм ҒмңјлЎң мәҗмӢұн•©лӢҲлӢӨ.

ML-25Mмқҳ 600MB train splitмқҖ мІ« epochмқҳ мІ« readм—җ лӘҮ мҙҲк°Җ кұёлҰ¬кі , мқҙнӣ„лЎңлҠ” ліё мҝЎл¶Ғмқҙ pq.read_tableлЎң мқјкҙ„ RAMм—җ лЎңл“ңн•ҳкё° л•Ңл¬ём—җ disk cache м—¬л¶ҖмҷҖ л¬ҙкҙҖн•ҙм§‘лӢҲлӢӨ.

multi-rank лҸҷмӢң мқҪкё° мӢң contention
